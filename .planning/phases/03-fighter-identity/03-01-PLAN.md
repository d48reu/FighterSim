def get_fighter_timeline(fighter_id: int) -> Optional[dict]:
    """Existing fight query pattern -- single query joining Fight + Event, ordered by date."""
    fights = session.execute(
        select(Fight, Event)
        .join(Event, Fight.event_id == Event.id)
        .where(or_(Fight.fighter_a_id == fighter_id, Fight.fighter_b_id == fighter_id),
               Fight.winner_id.isnot(None))
        .order_by(Event.event_date.asc(), Fight.id.asc())
    ).all()
